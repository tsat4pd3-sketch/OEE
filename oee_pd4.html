function initParts() { // Original part list const parts = [ /* ...existing parts... */ ]; // Exclude specific parts const excludedParts = ['bending gor', 'bending lwr bar']; const filteredParts = parts.filter(part => !excludedParts.includes(part)); // Initialize part filter with filtered parts const partFilter = document.getElementById('partFilter'); filteredParts.forEach(part => { const option = document.createElement('option'); option.value = part; option.text = part; partFilter.appendChild(option); }); }
